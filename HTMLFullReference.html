الـ Meta Tags وبنية الصفحة

<head> <meta> <link>
index.html — <head>
<head>
    <title>Home</title>
    <meta charset="utf-8">
    <meta name="description" content="هذا هو المحتوى">
    <meta name="keywords"    content="word1, word2">
    <meta name="author"      content="Hossam">
    <meta http-equiv="refresh" content="21">
    <meta http-equiv="refresh" content="20; url=https://www.google.com">
    <link rel="stylesheet" href="style.css">
</head>
الكودالشرح
charset="utf-8"اللغة اللي هتُكتب بيها الصفحة — ضروري للعربية
name="description"بتوصف الصفحة — لو حد بحث هيجيله المحتوى اللي فيها
name="keywords"الكلمات المفتاحية — لو المستخدم دخلها هتجيله الصفحة
name="author"اسم اللي كتب الصفحة
content="21"الصفحة هتعمل ريفرش بعد 21 ثانية
content="20; url=..."ريفرش وتحويل لصفحة تانية بعد 20 ثانية
link rel="stylesheet"ربط ملف CSS بالصفحة

الصور

<img>
img element
<img
    src="imege\th.jpg"
    alt="Image"
    width="200"
    height="200"
    border="5">
الخاصيةالشرح
src="imege\th.jpg"مكان الصورة — لو جمب الملف بنكتب اسمها بس، أو URL من الإنترنت
alt="Image"النص اللي بيظهر في حالة عدم تحميل الصورة
width / heightالأبعاد بالبكسل
border="5"حدود حول الصورة
العنصر ده مش بياخد السطر كله — inline element — ممكن نحط صورة من الإنترنت في src

الصوت

<audio> <source>
audio element
<audio controls loop muted autoplay>
    <source src="audio\azan.mp3" type="audio/mpeg">
    <source src="audio\azan.wav" type="audio/wav">
    <source src="audio\azan.ogg" type="audio/ogg">
    your browser does not support the audio tag
</audio>
الخاصيةالشرح
controlsبتظهر العنصر بالتحكم — عشان يظهر ملف الصوت
loopبيعيد تشغيل الملف الصوتي تلقائياً
mutedبيعمل الصوت بدون صوت
autoplayبيشغل الملف الصوتي تلقائياً عند فتح الصفحة
<source> × 3 امتداداتالتلات سطور بتلات امتدادات مختلفة — لو مشتغلش بأول امتداد يشتغل باللي بعده
type="audio/mpeg"بيحدد نوع الصوت للمتصفح
النص الأخيربيظهر لو المتصفح مش عارف يشغل الصوت

الفيديو

<video> <source> <track>
video element
<video
    controls loop muted autoplay
    width="10%"
    poster="imege\poster.jpg"
    preload="auto"
    preload="metadata"
    preload="none">

    <source src="vedios\vedio1.mp4"  type="video/mp4">
    <source src="vedio1.webm"        type="video/webm">
    <source src="vedios\vedio1.ogv"  type="video/ogg">
    your browser does not support the video tag

    <track src="vedio1.vtt" kind="subtitles" srclang="en" label="English">
    <track default src="vedio1.vtt" kind="subtitles" srclang="ar" label="Arabic">
</video>
الخاصيةالشرح
controlsبتظهر العنصر بالتحكم عشان يظهر ملف الفيديو
loopبيعيد تشغيل الفيديو تلقائياً
mutedبيشغله بدون صوت
autoplayبيشغله تلقائياً عند فتح الصفحة
width="10%"بيحدد العرض بالنسبة للصفحة
posterبيحدد صورة تظهر قبل تحميل الفيديو
preload="auto"بيحمّل الفيديو كله قبل تشغيله
preload="metadata"بيحمّل البيانات اللي بتظهر في العنصر بس
preload="none"بيحمّل الفيديو بعد ما المستخدم يشغله فقط
<source> × 3 امتداداتلو مشتغلش بالأول يشتغل باللي بعده
<track>بيحدد ملف الترجمة
kind="subtitles"بيحدد نوع الترجمة
srclang="ar"بيحدد لغة الترجمة
label="Arabic"بيحدد اسم الترجمة اللي بيظهر للمستخدم
defaultبيحدد إنها الترجمة الافتراضية

القوائم

<ul> <ol> <dl> <li> <dt> <dd>
ul — unordered list
<ul>
    <li>1</li>
    <li>2</li>
    <li>3</li>
</ul>
الكودالشرح
<ul>بتحدد إنها قائمة غير مرتبة — بتظهر نقاط
<li>بتحدد العناصر اللي هتدخلها في القائمة
ol — ordered list
<ol type="A" reversed start="4">
    <li>1</li>
    <li>2</li>
    <li>3</li>
</ol>
الخاصيةالشرح
<ol>بتحدد إنها قائمة مرتبة
type="A"بتحدد نوع القائمة — A = حروف كبيرة / a = صغيرة / 1 = أرقام / I = روماني
reversedبتحدد إنها تبدأ من الرقم الأخير
start="4"بتحدد الرقم اللي هتبدأ منه القائمة
dl — definition list
<dl>
    <dt>1</dt>
    <dd>2</dd>

    <dt>3</dt>
    <dd>4</dd>
</dl>
الكودالشرح
<dl>بتحدد إنها قائمة تعريفية
<dt>بتحدد العناصر اللي هتدخلها في القائمة (المصطلح)
<dd>بتحدد الوصف اللي بيظهر تحت العنصر (indent)

target="_parent"بيفتح الرابط في الصفحة اللي فوقه
target="_top"بيفتح الرابط في الصفحة اللي فوق كل الصفحات

mailto:بيفتح برنامج البريد الإلكتروني ويبعت على الميل ده
tel:بيفتح المتصفح للاتصال بالرقم ده
javascript:alert()بينفذ كود JavaScript — هيفتح نافذة منبثقة
download="photo"بيحدد اسم الملف اللي هيتنزل — لازم يشتغل على سيرفر — ممكن تحط أي حاجة مكان الفيديو وهتنزل

النماذج

<form> <label> <fieldset> <legend>
form element
<form action="test.php" method="post">

    <label for="TEXT">enter text</label>
    <input id="TEXT" type="text" placeholder="text" name="text">

    <label for="pass">enter password</label>
    <input id="pass" type="password" required placeholder="password">

    <label for="email">enter email</label>
    <input id="email" type="email" value="@gmail.com" placeholder="email">

    <input type="submit" value="send">

</form>

<fieldset>
    <legend>enter your data</legend>
    ...
</fieldset>
الكودالشرح
action="test.php"الملف اللي هيتم إرسال البيانات ليه
method="post"طريقة إرسال البيانات — بيخبيها من عنوان URL
<label for="TEXT">النص اللي بيظهر قبل الحقل — for بيتحط فيه id الحقل المرتبط بيه
placeholder="text"النص اللي بيظهر داخل الحقل كتلميح
name="text"اسم الحقل اللي هيتم إرسال البيانات منه للسيرفر
requiredلازم يدخل قيمة في الحقل — إجباري
value="@gmail.com"القيمة الافتراضية للحقل — ممكن المستخدم يعدل عليها
type="submit"زر الإرسال — بيرسل البيانات
<fieldset>إنشاء مربع حول مجموعة من الحقول
<legend>بتحدد العنوان اللي بيظهر فوق الحقول أو بين خط المربع

textarea element
<form novalidate>
    <textarea
        required
        maxlength="100"
        minlength="10"
        name="text"
        placeholder="text"
        cols="20"
        rows="10"
        autofocus>
    </textarea>
    <input type="submit" value="submit">
</form>
الخاصيةالشرح
novalidateبتحدد إنها مش هتتحقق من البيانات اللي داخلة
<textarea>حقل نصي كبير — متعدد الأسطر
requiredلازم يدخل قيمة في الحقل
maxlength="100"أقصى عدد حروف يمكن إدخالها
minlength="10"أقل عدد حروف يمكن إدخالها
nameاسم الحقل اللي هيتم إرسال البيانات منه
placeholderالنص اللي بيظهر داخل الحقل
cols="20"عدد الأعمدة اللي هتظهر في الحقل (العرض)
rows="10"عدد الصفوف اللي هتظهر في الحقل (الطول)
autofocusبتحدد إنها تكون الحقل الافتراضي المختار عند فتح الصفحة
🪟

الإطارات المضمّنة

<iframe> <object> <embed>
iframe — إضافة صفحات خارجية
<iframe
    src="https://www.wikipedia.org/"
    width="300"
    height="170"
    frameborder="0"
    scrolling="no">
</iframe>

<object data="https://www.wikipedia.org/"></object>

<embed src="https://www.wikipedia.org/"></embed>
الخاصيةالشرح
<iframe>إطار لموقع أو صفحة — من خلاله نقدر نضيف موقع أو فيديو أو حتى صفحة تانية عندنا
srcالموقع اللي هيتم عرضه في الإطار
width / heightعرض وطول الإطار
frameborder="0"بيخفي الحدود اللي حول الإطار
scrolling="no"بيخفي الشريط اللي بيظهر لو الموقع طويل
<object>نفس iframe
<embed>نفس iframe
iframe — تضمين فيديو يوتيوب
<iframe
    width="500"
    height="250"
    src="https://www.youtube.com/embed/o6As7cP3eIo?list=PLknwEmKsW8Ou...">
</iframe>
الكودالشرح
youtube.com/embed/رابط التضمين من يوتيوب — بتاخده من زر Share ثم Embed
ممكن نحط أي حاجة مكان الفيديو وهتنزل — iframe بيشتغل مع أي موقع أو فيديو أو PDF
